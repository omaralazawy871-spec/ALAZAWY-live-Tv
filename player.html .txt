    <a href="index.html" class="back-btn">العودة للقنوات</a>
    <video id="video" controls autoplay playsinline></video>
    
    <script>
        // استلام رابط البث من الصفحة الرئيسية
        const params = new URLSearchParams(window.location.search);
        const originalStreamUrl = params.get('stream');

        if (originalStreamUrl) {
            // استخدام بروكسي لتشغيل رابط السيرفر (HTTP) داخل موقعك الآمن (HTTPS)
            const videoSrc = 'https://corsproxy.io/?' + encodeURIComponent(originalStreamUrl);
            var video = document.getElementById('video');
            
            if (Hls.isSupported()) {
                var hls = new Hls();
                hls.loadSource(videoSrc);
                hls.attachMedia(video);
                hls.on(Hls.Events.MANIFEST_PARSED, function() {
                    video.play();
                });
            } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
                video.src = videoSrc;
                video.play();
            }
        }
    </script>
